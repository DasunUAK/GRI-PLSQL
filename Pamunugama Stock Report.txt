SELECT t.PART_NO,q.DESCRIPTION,t.SERIAL_NO,t.CF$_MANUFACTURED_DATE,t.RECEIPT_DATE,t.CF$_C_NET_WEIGHT,t.QTY_ONHAND,s.unit_cost

FROM
INVENTORY_PART_IN_STOCK_CFV t 
INNER JOIN Inventory_Part_Cfv q on t.PART_NO = q.PART_NO
INNER JOIN INVENTORY_PART_IN_STOCK_UIV s ON t.OBJKEY = s.objkey

where t.WAREHOUSE = 'PMG' AND t.CF$_LOCATION_GROUP = 'FGPK' AND t.QTY_ONHAND > 0 and t.SERIAL_NO <> '*'

Order by t.PART_NO
