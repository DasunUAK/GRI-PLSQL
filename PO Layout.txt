SELECT T0.Order_No,T0.Vendor_No,T1.name,T0.WANTED_RECEIPT_DATE,T2.LINE_NO,T2.PART_NO,T2.DESCRIPTION,T2.BUY_QTY_DUE,t2.FBUY_UNIT_PRICE,T2.FEE_CODE,T2.DISCOUNT,(T2.FBUY_UNIT_PRICE*T2.BUY_QTY_DUE) AS "Total Value"
FROM Purchase_Order T0 
INNER JOIN Supplier_Info_General T1 ON T0.VENDOR_NO = T1.supplier_id 
INNER JOIN Purchase_Order_Line_Part T2 ON T0.ORDER_NO = T2.ORDER_NO

WHERE T0.Order_No = 340
